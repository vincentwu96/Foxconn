select si.MFG_NAME||'-'||si.CITY as MFG ,fa.FAID ,CREATEDBY ,CREATIONDATE ,STATUS ,OWNER ,'HIDDEN' as ASSOCIATEDTEST ,GUTI ,TESTSTARTTIME ,
case when CATEGORY_CODE IS null then '-' else CATEGORY_CODE end as CATEGORY ,case when CATEGORY_FAULTCODE IS null then '-' else CATEGORY_FAULTCODE end as CODE ,
case when CATEGORY_TYPECODE IS null then '-' else CATEGORY_TYPECODE end as TYPE ,case when STATUSCOMMENT IS null then 'empty' else to_char(STATUSCOMMENT) end as STATUSCOMMENT ,
SYSTEMFAMILY ,SYSTEMSERIALNUMBER ,FAILEDLOCATION ,FAILEDLOGOP ,case when TESTSETNAME IS null then '-' else TESTSETNAME end as TESTSETNAME from SITEINFO si,
FA fa LEFT OUTER JOIN FA_TO_TESTSET fatts ON fa.faid = fatts.faid where fa.CREATIONDATE between TO_DATE('10-01-2019 00:00:01','DD-MM-YYYY HH24:MI:SS')
AND TO_DATE('01-03-2019 23:59:59','DD-MM-YYYY HH24:MI:SS') AND fa.SYSTEMFAMILY in ('E1-2c','ODA_X7-2','X7-2','X7-2L','X7-2c','X7-8','ZS7-2') AND fa.FAILEDLOGOP in 
('BBT','FSC','SFT','FST') AND fa.STATUS in ('ANALYSIS','CLOSED','OPEN','REPAIR','VERIFY') AND fa.SITEID in (92) AND fa.SITEID = si.SITEID ORDER BY 1,2,3,4 DESC
